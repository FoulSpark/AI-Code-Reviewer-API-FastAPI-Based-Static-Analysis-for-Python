Stage 1 — Understand
gemini-1.5-pro · JSON out
UNDERSTAND_PROMPT = """ You are a code specification extractor. Analyse the user request and output ONLY valid JSON. No markdown fences. No explanation. JSON only. Schema: { "task": "one sentence", "language": "python | javascript | typescript | ...", "input_description": "what the code receives", "output_description": "what the code produces", "constraints": ["list of rules or edge cases"], "clarifications_needed": [] } If the request is ambiguous, populate clarifications_needed and leave other fields as empty strings. USER REQUEST: {user_request} PREVIOUS ERRORS (if any): {errors} """
errors is injected on retry so the model knows what it got wrong last time. Pass empty string on first call.
Stage 2 — Plan
gemini-1.5-pro · JSON out
PLAN_PROMPT = """ You are a code architect. Produce a plan and a complete symbol table. This symbol table is a CONTRACT — the implementer will only use names you define here. SPEC: {spec_json} PREVIOUS SEMANTIC ERRORS (if replanning): {errors} Output ONLY valid JSON. No markdown. No explanation. { "pseudocode": ["step 1", "step 2", ...], "symbol_table": { "variables": [ {"name": "rows", "type": "list[dict]", "purpose": "parsed CSV rows"} ], "functions": [ { "name": "load_csv", "params": [{"name": "path", "type": "str"}], "returns": "list[dict]", "purpose": "reads CSV and returns list of row dicts", "stub": "def load_csv(path: str) -> list[dict]:\n ..." } ], "classes": [] }, "imports": ["csv", "pathlib"] } Rules: - Every variable used in implementation MUST appear in variables - Every function MUST appear in functions with a complete stub - Do NOT write implementation — stubs only """
Stage 3 — Implement
gemini-1.5-pro · raw code out
IMPLEMENT_PROMPT = """ You are a code writer. Implement the plan exactly. SPEC: {spec_json} PLAN + SYMBOL TABLE: {plan_json} PREVIOUS ERRORS TO FIX (if retrying): {errors} Rules: - Use ONLY variable names from symbol_table.variables - Implement EVERY function in symbol_table.functions - Use ONLY imports from the plan - Do NOT invent new top-level names not in the symbol table - Output raw {language} code only - No markdown fences, no explanation, no comments """
The "no comments" rule matters — comments can confuse the syntax checker and bloat token count on retries.
Stage 4a — Syntax (LLM fallback only)
gemini-1.5-flash · JSON out · skipped for Python
SYNTAX_PROMPT = """ You are a {language} syntax validator. Check the code ONLY for structural syntax errors: - Missing colons after if / for / while / def / class - Unclosed brackets, parens, or braces - Invalid indentation - Unterminated string literals Do NOT check logic, variable names, or semantics. Output ONLY valid JSON. No markdown. No explanation. { "passed": true, "errors": [] } OR if failed: { "passed": false, "errors": [ {"line": 7, "issue": "missing colon after if", "fix": "add : at end of line 7"} ] } CODE: {code} """
Stage 4b — Semantic
gemini-1.5-flash · JSON out · always runs
SEMANTIC_PROMPT = """ You are a {language} semantic validator. SYMBOL TABLE (every name that must exist): {symbol_table_json} Check the code for: 1. Variables used but not assigned or not in symbol_table 2. Functions called but not defined or not in symbol_table 3. Wrong argument count passed to a function 4. Obvious type mismatches visible from context For each error, set resolution to: "fix_in_code" — typo or minor fix, implementer can patch it "add_to_plan" — symbol is genuinely missing from the table, the plan must be revised first Output ONLY valid JSON. No markdown. No explanation. { "passed": true, "errors": [] } OR: { "passed": false, "errors": [ { "issue": "undefined variable 'result'", "resolution": "fix_in_code", "detail": "result used on line 12 but never assigned" } ] } CODE: {code} """
Full async orchestrator
pipeline.py — complete file
import ast, asyncio, json, re import google.generativeai as genai genai.configure(api_key="YOUR_KEY") pro = genai.GenerativeModel("gemini-1.5-pro") flash = genai.GenerativeModel("gemini-1.5-flash") MAX_PLAN_RETRIES = 3 MAX_IMPL_RETRIES = 3 # ── helpers ─────────────────────────────────────────────── async def call(model, prompt: str) -> str: loop = asyncio.get_event_loop() r = await loop.run_in_executor( None, lambda: model.generate_content(prompt) ) return r.text.strip() def parse(text: str) -> dict: clean = re.sub(r"```(?:json)?|```", "", text).strip() return json.loads(clean) def fmt_errors(errors) -> str: if not errors: return "none" return json.dumps(errors, indent=2) # ── stage 1 ─────────────────────────────────────────────── async def understand(user_request: str) -> dict: p = UNDERSTAND_PROMPT.format( user_request=user_request, errors="none" ) return parse(await call(pro, p)) # ── stage 2 ─────────────────────────────────────────────── async def plan(spec: dict, errors=None) -> dict: p = PLAN_PROMPT.format( spec_json=json.dumps(spec, indent=2), errors=fmt_errors(errors) ) return parse(await call(pro, p)) # ── stage 3 ─────────────────────────────────────────────── async def implement(spec: dict, plan_obj: dict, errors=None) -> str: p = IMPLEMENT_PROMPT.format( spec_json=json.dumps(spec, indent=2), plan_json=json.dumps(plan_obj, indent=2), errors=fmt_errors(errors), language=spec["language"] ) return await call(pro, p) # ── stage 4a — native for Python, flash for others ──────── def syntax_native(code: str) -> dict: try: ast.parse(code) return {"passed": True, "errors": []} except SyntaxError as e: return {"passed": False, "errors": [ {"line": e.lineno, "issue": str(e), "fix": "check syntax near this line"} ]} async def syntax_llm(code: str, lang: str) -> dict: p = SYNTAX_PROMPT.format(language=lang, code=code) return parse(await call(flash, p)) async def check_syntax(code: str, lang: str) -> dict: if lang == "python": return syntax_native(code) # instant, no LLM call return await syntax_llm(code, lang) # ── stage 4b ────────────────────────────────────────────── async def check_semantic(code: str, plan_obj: dict, lang: str) -> dict: p = SEMANTIC_PROMPT.format( language=lang, symbol_table_json=json.dumps( plan_obj["symbol_table"], indent=2), code=code ) return parse(await call(flash, p)) # ── main pipeline ───────────────────────────────────────── async def run(user_request: str) -> str: spec = await understand(user_request) lang = spec["language"] plan_obj = await plan(spec) for plan_try in range(MAX_PLAN_RETRIES): impl_errors = None for impl_try in range(MAX_IMPL_RETRIES): code = await implement(spec, plan_obj, impl_errors) # fire both checks concurrently syn, sem = await asyncio.gather( check_syntax(code, lang), check_semantic(code, plan_obj, lang) ) if syn["passed"] and sem["passed"]: return code # done! needs_replan = any( e["resolution"] == "add_to_plan" for e in sem.get("errors", []) ) if needs_replan: # break inner → replan with semantic errors plan_obj = await plan( spec, sem["errors"] ) break # fix_in_code: retry implement with all errors impl_errors = ( syn.get("errors", []) + sem.get("errors", []) ) raise RuntimeError("Pipeline exhausted retries") # ── entry point ─────────────────────────────────────────── if __name__ == "__main__": result = asyncio.run(run( "Read a CSV, return top 5 rows by 'score' descending" )) print(result)

Retry cost model
Each retry adds ~3-4s (one Pro call for Implement, or ~2s if only Flash checks). With MAX_IMPL_RETRIES=3 and MAX_PLAN_RETRIES=3, worst case is ~40s. In practice symbol-table-driven planning reduces retries to 0-1 in >80% of cases.
Implement retry (+fix_in_code)
+3s per retry (1 Pro call)
Plan retry (+add_to_plan)
+5s per retry (1 Pro + full inner loop)
Further optimisations (optional)
Streaming output Use stream=True on Implement so code appears incrementally. Does not reduce total time but feels faster.
Cache Understand If user re-runs same request, skip Stage 1 by hashing the input string.
pylint / mypy for semantic Replace Stage 4b with subprocess call to pylint --errors-only. Faster and more accurate than Flash for Python.
Context caching Gemini supports prompt caching. Cache the system portion of Plan and Implement prompts to reduce token cost on retries.
